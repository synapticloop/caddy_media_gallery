% Preamble for the Caddy Media Gallery docs PDF.
% Included via pandoc's --include-in-header=preamble.tex
%
% Note: the font paths use `./fonts/` (relative to the docs
% directory, since pandoc is run from there by build-docs.sh).
% This keeps the preamble portable — no user-specific absolute
% paths in the committed file.

\usepackage{graphicx}
\usepackage{xcolor}

% Google Fonts (variable fonts from docs/fonts/)
% Libre Baskerville for body text
% JetBrains Mono for code blocks
\usepackage{fontspec}
\setmainfont{LibreBaskerville-VF}[
  Path = ./fonts/,
  Extension = .ttf,
  UprightFont    = LibreBaskerville-VF,
  BoldFont       = LibreBaskerville-VF,
  ItalicFont     = LibreBaskerville-Italic-VF,
  BoldItalicFont = LibreBaskerville-Italic-VF
]
\setmonofont{JetBrainsMono-VF}[
  Path = ./fonts/,
  Extension = .ttf,
  UprightFont    = JetBrainsMono-VF,
  BoldFont       = JetBrainsMono-VF,
  ItalicFont     = JetBrainsMono-Italic-VF,
  BoldItalicFont = JetBrainsMono-Italic-VF
]
\setmathfont{Latin Modern Math}

% Light grey for code highlighting (inline + blocks)
\definecolor{codebg}{HTML}{F0F0F0}
\definecolor{codeborder}{HTML}{000000}

% Inline code: pandoc emits \texttt{code} for backtick-wrapped
% spans in markdown. Wrap that in a light-grey colorbox so
% inline `code` reads as a subtle highlight, not just a
% different font.
\let\oldtexttt\texttt
\renewcommand{\texttt}[1]{\colorbox{codebg}{\oldtexttt{#1}}}

% Code blocks (longer snippets). The pandoc command uses
% --listings so fenced code blocks become \begin{lstlisting}.
% Settings per the user spec (2026-06-17):
%   - light grey background
%   - black outline (frame=single, rulecolor=black)
%   - line numbers on the LEFT, in monospace
\usepackage{listings}
\lstset{
  basicstyle=\ttfamily\small,
  backgroundcolor=\color{codebg},
  frame=single,
  rulecolor=\color{codeborder},
  numbers=left,
  numberstyle=\ttfamily\small,
  numbersep=10pt,
  xleftmargin=15pt,
  framexleftmargin=5pt,
  breaklines=true,
  showstringspaces=false,
  aboveskip=10pt,
  belowskip=10pt,
  columns=fullflexible,
  keepspaces=true,
}
